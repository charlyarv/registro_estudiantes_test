CREATE TABLE `redefectiva`. `estudiantes`
( `folio_id` INT NOT NULL AUTO_INCREMENT , 
`matricula` VARCHAR(50) NULL DEFAULT NULL , 
`nombre` VARCHAR(100) NULL DEFAULT NULL , 
`genero` VARCHAR(50) NULL DEFAULT NULL , 
`grado_escolar` INT(5) NULL DEFAULT NULL , 
`estatus` INT(5) NOT NULL DEFAULT '1' , 
`created_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ,
`updated_at` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ,
PRIMARY KEY (`folio_id`)) 
ENGINE = InnoDB;
